CREATE TABLE produtos (
    id VARCHAR(50) PRIMARY KEY, -- Identificador único da tabela
    codigo VARCHAR(50) NOT NULL, -- Código do produto
    descricao VARCHAR(255) NOT NULL, -- Descrição do produto
    departamento VARCHAR(100), -- Departamento ao qual o produto pertence
    preco DECIMAL(10, 2) NOT NULL, -- Preço do produto com duas casas decimais
    status BOOLEAN NOT NULL DEFAULT TRUE, -- Status do produto (ativo/inativo)
    acoes INT NOT NULL -- (acoes IN ('Criar', 'Atualizar', 'Excluir')) -- Enumeração para ações
);
